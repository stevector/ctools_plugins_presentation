function abbreviated_ctools_node_type_ctools_access_summary($conf, $context) {
  if (!isset($conf['type'])) {
    $conf['type'] = array();
  }
  $types = node_type_get_types();
  $names = array();

  foreach (array_filter($conf['type']) as $type) {
    if (!empty($types[$type])) {
      $names[] = check_plain($types[$type]->name);
    }
  }

  return t('@identifier is any node type', array('@identifier' => $context->identifier));
}

  $node = node_load($arg);
  if (!$node) {
    return FALSE;
  }

  return ctools_context_create('node', $node);
}

Context Plugins

Context Plugins

A wrapper around a variable

function abbreviated_ctools_context_create_node($empty, $data = NULL, $conf = FALSE) {
  $context = new ctools_context('node');
  $context->plugin = 'node';

  $nid = $data['nid'];
  $data = node_load($nid);

  $context->data     = $data;
  $context->title    = $data->title;
  $context->argument = $data->nid;
  $context->restrictions['type'] = array($data->type);
  return $context;
}

function abbreviated_ctools_node_access_ctools_access_check($conf, $context) {
  list($user_context, $node_context) = $context;

  if ($conf['type'] == 'create') {
    return node_access('create', $node_context->data->type, $user_context->data);
  }
  else {
    return node_access($conf['type'], $node_context->data, $user_context->data);
  }
}

function abbreviated_ctools_node_title_content_type_render($subtype, $conf, $panel_args, $context) {
  if (empty($context) || empty($context->data)) {
    return;
  }

  // Get a shortcut to the node.
  $node = $context->data;

  // Build the content type block.
  $block = new stdClass();
  $block->content = check_plain($node->title);
  return $block;
}

Child plugins

Here's where the foreach {} goes

function abbreviated_ctools_entity_bundle_ctools_access_get_children($plugin, $parent) {
  $entities = entity_get_info();
  foreach ($entities as $entity_type => $entity) {
    $plugin['title'] = t('@entity: Bundle', array('@entity' => $entity['label']));
    $plugin['name'] = $parent . ':' . $entity_type;
    $plugin['required context'] = new ctools_context_required(t(ucfirst($entity_type)), $entity_type);
    $plugins[$parent . ':' . $entity_type] = $plugin;
  }
  return $plugins;
}

function custom_ctools_plugin_post_alter(&$plugin, &$info) {

  // Override two plugin functions.
  if ($plugin['content types'] === 'fieldable_panels_panes_fieldable_panels_pane_content_type_content_types') {
    $plugin['content types'] = 'custom_fieldable_panels_pane_content_type_content_types';
  }

  if ($plugin['context'] === 'ctools_context_create_entity') {
    $plugin['context'] = 'custom_ctools_context_create_entity';
  }
}
